% Use memoir for vplace
\documentclass[12pt]{memoir} 
\usepackage[left=2cm,right=2cm,top=6cm,bottom=5cm]{geometry}
\usepackage[latin1]{inputenc} % required latin1 for characters like: ô ç etc.

\usepackage{wallpaper} % For background image frame

\usepackage{datatool} % For loading data from file (e.g. from csv)
% Load database 'names' from file 'namelist.csv'
\DTLloaddb{names}{namelist.csv}

\begin{document}
\linespread{4}\selectfont
\pagestyle{empty}
% Set "cern-webfest-background.pdf" as background image
\TileWallPaper{\paperwidth}{\paperheight}{cern-webfest-background}

% For each record (line) in database 'names'
% Assign field values by name to macros
\DTLforeach{names}
{\name=Name, \gender=Gender}{
  {
    \begin{vplace}[0.2]
    % Insert '\name' field
    {\LARGE\bfseries\MakeUppercase{\name}}
    \end{vplace}
  }
  \pagebreak
}

\end{document}
